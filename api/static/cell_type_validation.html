{
  "note": "Project note",
  "group": "Project group",
  "source": "Neu3",
  "task list": [
    {
      "task type": "type review",
      "task result id": "1173408617_5813021469",
      "body ID A": 1173408617,
      "body ID B": 5813021469,
      "match_score": 0.27,
      "debug": {
        "original_uuid": "34d7acb0167d438baa85c47ea0a739d4",
        "comment": "initial typereview set"
      }
    },
    {
      "task type": "type review",
      "task result id": "1067598881_1141682748",
      "body ID A": 1067598881,
      "body ID B": 1141682748,
      "match_score": 0.298,
      "debug": {

Elements in gray are optional, all others are required. Any additional data not shown above is ignored. "task result id" values are in the format body ID_body ID.
